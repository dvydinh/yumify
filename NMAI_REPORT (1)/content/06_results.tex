\subsection{Phân tích Chi tiết}

\subsubsection{NLP: Ontology-based NER}
\begin{itemize}
  \item \textbf{Budget parsing:} Hỗ trợ \texttt{\$10}, \texttt{budget 15}, \texttt{for \$8}. Độ chính xác: 100\% trên test cases.
  \item \textbf{Cuisine detection:} ML Naive Bayes là người ra quyết định chính. Fallback: keyword matching.
  \item \textbf{Health NER:} Nhận diện stomachache, gout, diabetes, hypertension, seafood allergy, lactose/gluten intolerance.
  \item \textbf{Hạn chế:} Chưa xử lý tốt câu mơ hồ (``I want something good'') --- confidence thấp.
\end{itemize}

\subsubsection{ML: Naive Bayes Cuisine Classification}
\begin{itemize}
  \item \textbf{Tập local (25 mẫu):} 3/5 predictions đúng. \texttt{kimchi, gochujang} $\to$ Japanese (sai, nên là Korean) --- do chỉ có 2 mẫu Korean.
  \item \textbf{Tập Kaggle (5000 mẫu):} Stratified sampling đảm bảo $\leq$ 625 mẫu/cuisine, giải quyết class imbalance.
  \item \textbf{Log-space computation:} Tránh underflow khi nhân nhiều xác suất nhỏ.
  \item \textbf{Hạn chế:} Conditional independence assumption --- không capture tương quan giữa nguyên liệu.
\end{itemize}

\subsubsection{A*: Tính Tối ưu và Đa dạng}
\begin{itemize}
  \item \textbf{Admissibility:} $h(S) = (N - \text{day}) \times \min\_\text{cost}$ luôn $\leq h^*(S)$.
  \item \textbf{Canonical State:} Sorted tuple gộp hoán vị, giảm $O(R^N) \to O(R^N / N!)$.
  \item \textbf{Repetition Penalty:} Exponential penalty ($\text{cost} \times 2^k$) + consecutive penalty ($\text{cost} \times 3$) đảm bảo meal plan đa dạng.
  \item \textbf{Kết quả:} 3-day \$50 $\to$ 3/3 unique, 10 nodes. 5-day \$80 $\to$ 5/5 unique.
  \item \textbf{Trước penalty:} A* chọn ``Chicken Enchilada Casserole $\times$ 3'' (rẻ nhất, \$5.47/món). Vô nghĩa thực tế.
\end{itemize}

\subsubsection{CSP: Dual-constraint Pruning}
\begin{itemize}
  \item 2 ràng buộc: Budget ($C_1$), Calories ($C_2$).
  \item \textbf{MRV heuristic} chọn biến có domain nhỏ nhất trước.
  \item \textbf{Bounds Propagation} tính min/max future calories để cắt sớm.
  \item \textbf{Kết quả:} 8 backtracks cho Spicy Arrabiata Penne, success.
\end{itemize}

\subsubsection{Bayesian Risk: Extended Noisy-OR}
\begin{itemize}
  \item 24 risk factors (English CPT) + 8 synergy pairs.
  \item \textbf{Spicy Shrimp Oil Pasta} (stomachache): $P = 0.9970$, 9 factors detected (chili, oil, seafood, + 4 synergy pairs).
  \item \textbf{Steamed Vegetables}: $P = 0.0000$, 0 factors --- baseline an toàn.
  \item $P(\text{risk})$ luôn $\in [0, 1]$ theo lý thuyết Noisy-OR, không cần clamping.
\end{itemize}

\subsection{Hạn chế}

\begin{enumerate}
  \item \textbf{ML trên tập local:} 25 mẫu không đủ để phân biệt Korean vs Japanese.
  \item \textbf{Cost ước tính:} Dựa trên cross-reference heuristic, không phải giá thị trường thực tế.
  \item \textbf{Kaggle dependency:} Nếu \texttt{kagglehub} timeout, fallback về 25 recipes.
  \item \textbf{Bayes CPT:} Xác suất gán bởi chuyên gia (subjective), chưa validated bằng data lâm sàng.
  \item \textbf{NLP:} Chỉ hỗ trợ tiếng Anh. Câu mơ hồ hoặc ngữ pháp sai cho confidence thấp.
\end{enumerate}
