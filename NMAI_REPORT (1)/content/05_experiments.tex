\section{Thực nghiệm}
\label{sec:experiments}

\subsection{Môi trường Thực nghiệm}
\begin{itemize}
  \item \textbf{Platform:} Google Colab (Python 3.10, RAM 12GB)
  \item \textbf{Libraries:} \texttt{numpy}, \texttt{requests} (Pure Python --- không dùng thư viện AI nặng)
  \item \textbf{Dataset chính:} Kaggle Food.com (5000 recipes qua stratified sampling, 15+ cuisines)
  \item \textbf{Dataset fallback:} \texttt{recipes.json} (25 recipes, 6 cuisines), \texttt{ingredients.json} (75 nguyên liệu)
\end{itemize}

\subsection{Phân tích Khám phá Dữ liệu (EDA)}

\subsubsection{Tập cục bộ (\texttt{recipes.json})}
\begin{itemize}
    \item 25 công thức, 6 cuisines: Italian (6), Japanese (5), Mexican (5), Western (4), Vietnamese (3), Korean (2)
    \item Cost range: \$0.99 --- \$40.28, mean \$17.28
    \item 100\% recipes có trường \texttt{cost} (đã patch cross-reference \texttt{ingredients.json})
    \item TF-IDF vocabulary: 759 terms, ma trận 25 $\times$ 759
\end{itemize}

\subsubsection{Tập Kaggle (Food.com)}
\begin{itemize}
    \item 230,000+ dòng CSV gốc, parse tối đa 5000 recipes
    \item Stratified sampling: MAX\_PER\_CUISINE = 625, đảm bảo không quá 12.5\% cho mỗi cuisine
    \item 3-tier cuisine detection: tag $\to$ recipe name (120+ patterns) $\to$ ingredient signature (80+ patterns)
    \item Heuristic quantity estimation: Thịt 200g, Hải sản 150g, Rau 100g, Gia vị 5g
    \item Cost computation: cross-reference \texttt{ingredients.json}, exact $\to$ substring $\to$ category fallback
\end{itemize}
